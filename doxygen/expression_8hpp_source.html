482  // Updates the list of active nodes
483  void update_active()
484  {
485  assert(m_x.size() == m_lb.size());
486 
487  // First we update the active nodes
488  std::vector<unsigned int> current(m_m), next;
489  m_active_nodes.clear();
490  // At the beginning, current contains only the output nodes connections
491  for (auto i = 0u; i < m_m; ++i) {
492  current[i] = m_x[(m_arity + 1) * m_r * m_c + i];
493  }
494  do
495  {
496  m_active_nodes.insert(m_active_nodes.end(), current.begin(), current.end());
497 
498  for (auto node_id : current)
499  {
500  if (node_id >=m_n) // we insert the input nodes connections as they do not have any
501  {
502  for (auto i = 1u; i <= m_arity; ++i) {
503  next.push_back(m_x[(node_id - m_n) * (m_arity + 1) + i]);
504  }
505  }
506  else{
507  m_active_nodes.push_back(node_id);
508  }
509  }
510  // We remove duplicates to avoid processing them and thus having a 2^N complexity
511  std::sort( next.begin(), next.end() );
512  next.erase( std::unique( next.begin(), next.end() ), next.end() );
513  current = next;
514  next.clear();
515  } while (current.size() > 0);
516 
517  // We remove duplicates and keep m_active_nodes sorted
518  std::sort( m_active_nodes.begin(), m_active_nodes.end() );
519  m_active_nodes.erase( std::unique( m_active_nodes.begin(), m_active_nodes.end() ), m_active_nodes.end() );
520 
521  // Then the active genes
522  m_active_genes.clear();
523  for (auto i = 0u; i<m_active_nodes.size(); ++i)
524  {
525  if (m_active_nodes[i] >= m_n)
526  {
527  unsigned int idx = (m_active_nodes[i] - m_n) * (m_arity + 1);
528  for (auto j = 0u; j <= m_arity; ++j) {
529  m_active_genes.push_back(idx + j);
530  }
531  }
532  }
533  for (auto i = 0u; i<m_m; ++i)
534  {
535  m_active_genes.push_back(m_r * m_c * (m_arity + 1) + i);
536  }
537  }
